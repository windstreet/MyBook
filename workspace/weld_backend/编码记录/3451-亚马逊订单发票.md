def run(uri=None):
    """子账号-税号"""

    wb = openpyxl.load_workbook(uri)

    account_res = dict()
    account = wb[u'子账号-税号']
    for i, item in enumerate(account.iter_rows()):
        if i == 0:
            continue
        else:
            account_res[item[0].value] = item[1].value
    return account_res
